CREATE TABLE Book (
 NUM NUMBER(5) PRIMARY KEY, 
 Category NVARCHAR2(1000),
 Title NVARCHAR2(1000), 
 Price NVARCHAR2(20), 
 Summary VARCHAR2(4000),
 Author NVARCHAR2(1000), 
 Pub NVARCHAR2(1000),
 Grade NVARCHAR2(10)
);

CREATE SEQUENCE Book_seq
INCREMENT BY 1
START WITH 1;

SELECT * FROM Book;

DESC Book;
